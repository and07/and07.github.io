/* Create the FileSystemObject */
var fso = new ActiveXObject("Scripting.FileSystemObject");

var tf = fso.CreateTextFile("dir_files.txt", true);
/* Point the Object to a specific folder */
var fsofolder = fso.GetFolder("C:\\");
/* get a collection of the files contained in that folder */
var colFiles  = fsofolder.Files;
/* Create an Enumerator so that we can move through the collection */
var fc = new Enumerator( colFiles );
/* Create a variable to store an output message in */
var msg = "";
/* Loop through the Enumerator and add each item to our variable */
for (; !fc.atEnd(); fc.moveNext() ){
    msg += fc.item() + "\n";
    tf.WriteLine(fc.item());
}
/* Show the user the results */
WScript.Echo( msg );
tf.Close();
WScript.Echo( 'DONE' );

# Python 3.3.5:

wget  ${WEBSITE_URL}  --header "Referer: www.python.org"

tar xvf Python-3.5.2.tar.xz
cd Python-3.5.2
./configure --prefix=/usr/local --enable-shared LDFLAGS="-Wl,-rpath /usr/local/lib"
make && make altinstall
# First get the setup script for Setuptools:

python3.5 ez_setup.py
easy_install-3.5 pip
pyvenv-3.5 python_script
# Activate the python_script sandbox and check the version of the default Python interpreter in it:
source python_script/bin/activate

pip3 install numpy

rm -rf Python-3.5.2
rm -rf Python-3.5.2.tar.xz

#python --version
# This will show Python 3.3.5
